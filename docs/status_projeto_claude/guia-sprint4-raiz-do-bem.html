GlobalExceptionMapper.java
package br.com.raizdobem.api.exception;

import jakarta.ws.rs.core.Response;
import jakarta.ws.rs.ext.ExceptionMapper;
import jakarta.ws.rs.ext.Provider;
import java.util.Map;

public class GlobalExceptionMapper {

    @Provider
    public static class NaoEncontradoMapper
            implements ExceptionMapper<RecursoNaoEncontradoException> {
        @Override
        public Response toResponse(RecursoNaoEncontradoException e) {
            return Response.status(Response.Status.NOT_FOUND)
                    .entity(Map.of("erro", e.getMessage()))
                    .build();
        }
    }

    @Provider
    public static class ValidacaoMapper
            implements ExceptionMapper<ValidacaoException> {
        @Override
        public Response toResponse(ValidacaoException e) {
            return Response.status(Response.Status.BAD_REQUEST)
                    .entity(Map.of("erro", e.getMessage()))
                    .build();
        }
    }

    @Provider
    public static class NegocioMapper
            implements ExceptionMapper<RegraDeNegocioException> {
        @Override
        public Response toResponse(RegraDeNegocioException e) {
            return Response.status(422)
                    .entity(Map.of("erro", e.getMessage()))
                    .build();
        }
    }

    @Provider
    public static class GenericoMapper
            implements ExceptionMapper<Exception> {
        @Override
        public Response toResponse(Exception e) {
            return Response.status(500)
                    .entity(Map.of("erro", "Erro interno: " + e.getMessage()))
                    .build();
        }
    }
}

import br.com.raizdobem.api.exception.ValidacaoException;
import br.com.raizdobem.api.exception.RegraDeNegocioException;
import java.util.Optional;

// Adicione este método na classe DentistaService:
public boolean validarCro(String cro) {
    // Validação 1: campo obrigatório
    if (cro == null || cro.isBlank()) {
        throw new ValidacaoException("CRO é obrigatório.");
    }

    // Validação 2: formato UF-NNNNN (ex: SP-12345)
    if (!cro.matches("^[A-Z]{2}-\\d{4,6}$")) {
        throw new ValidacaoException(
            "CRO inválido. Formato esperado: UF-NNNNN (ex: SP-12345). Recebido: " + cro
        );
    }

    // Validação 3: CRO único no banco
    Optional<DentistaDTO> existente = repository.find("croDentista", cro).firstResultOptional();
    if (existente.isPresent()) {
        throw new RegraDeNegocioException(
            "CRO '" + cro + "' já está cadastrado no sistema."
        );
    }

    return true;
}

// Adicione também o criar() que usa validarCro:

    validarCro(dto.getCroDentista()); // chama a validação

Adicione em PedidoAjudaService.java — o método mais complexo e mais valorizado. Gerencia transição de status com regras de negócio.

PedidoAjudaService.java — adicionar método
// Adicione estes imports:
import br.com.raizdobem.api.exception.*;
import br.com.raizdobem.api.entity.StatusPedido;
import br.com.raizdobem.api.repository.DentistaRepository;
import jakarta.inject.Inject;
import jakarta.transaction.Transactional;
import java.time.LocalDate;

// Adicione no topo da classe:
@Inject
DentistaRepository dentistaRepository;

// Método principal:
@Transactional
public PedidoAjudaDTO processarPedido(Long id, String novoStatus, Long idDentista) {

    // 1. Buscar pedido — lança 404 se não existir
    PedidoAjudaDTO pedido = repository.findById(id);
    if (pedido == null) {
        throw new RecursoNaoEncontradoException("PedidoAjuda", id);
    }

    // 2. Validar se o status informado é válido
    StatusPedido statusEnum;
    try {
        statusEnum = StatusPedido.valueOf(novoStatus.toUpperCase());
    } catch (IllegalArgumentException e) {
        throw new ValidacaoException(
            "Status inválido: '" + novoStatus + "'. Valores válidos: PENDENTE, APROVADO, REJEITADO"
        );
    }

    // 3. Regra: só PENDENTE pode ser APROVADO
    if (statusEnum == StatusPedido.APROVADO
            && pedido.getStatus() != StatusPedido.PENDENTE) {
        throw new RegraDeNegocioException(
            "Apenas pedidos PENDENTES podem ser APROVADOS. Status atual: "
            + pedido.getStatus()
        );
    }

    // 4. Regra: se aprovado com dentista, vincula
    if (statusEnum == StatusPedido.APROVADO && idDentista != null) {
        DentistaDTO dentista = dentistaRepository.findById(idDentista);
        if (dentista == null) {
            throw new RecursoNaoEncontradoException("Dentista", idDentista);
        }
        pedido.setDentistaDTO(dentista);
    }

    // 5. Regra: pedido REJEITADO não pode ser reaberto
    if (pedido.getStatus() == StatusPedido.REJEITADO) {
        throw new RegraDeNegocioException(
            "Pedido REJEITADO não pode ter seu status alterado."
        );
    }

    pedido.setStatus(statusEnum);
    return pedido; // @Transactional persiste automaticamente
}

// Adicione imports:
import br.com.raizdobem.api.exception.*;
import br.com.raizdobem.api.entity.StatusPedido;
import br.com.raizdobem.api.repository.PedidoAjudaRepository;
import br.com.raizdobem.api.repository.ProgramaRepository;

// Injete repositórios extras:
@Inject
PedidoAjudaRepository pedidoRepository;

@Inject
ProgramaRepository programaRepository;

// Método:
@Transactional
public BeneficiarioDTO adicionarDePedido(Long idPedido, Long idPrograma) {

    // 1. Validar pedido existe
    PedidoAjudaDTO pedido = pedidoRepository.findById(idPedido);
    if (pedido == null) {
        throw new RecursoNaoEncontradoException("PedidoAjuda", idPedido);
    }

    // 2. Regra: pedido precisa estar APROVADO
    if (pedido.getStatus() != StatusPedido.APROVADO) {
        throw new RegraDeNegocioException(
            "O pedido precisa estar APROVADO para gerar um beneficiário. Status atual: "
            + pedido.getStatus()
        );
    }

    // 3. Regra: um pedido gera apenas um beneficiário
    long existente = repository.count("pedido.id", idPedido);
    if (existente > 0) {
        throw new RegraDeNegocioException(
            "Já existe um beneficiário vinculado ao pedido de id " + idPedido
        );
    }

    // 4. Validar programa social
    ProgramaSocialDTO programa = programaRepository.findById(idPrograma);
    if (programa == null) {
        throw new RecursoNaoEncontradoException("ProgramaSocial", idPrograma);
    }

    // 5. Criar beneficiário com dados do pedido
    BeneficiarioDTO novo = new BeneficiarioDTO();
    novo.setCpf(pedido.getCpf());
    novo.setNomeCompleto(pedido.getNomeCompleto());
    novo.setDataNascimento(pedido.getDataNascimento());
    novo.setTelefone(pedido.getTelefone());
    novo.setEmail(pedido.getEmail());
    novo.setPedido(pedido);
    novo.setProgramaSocialDTO(programa);
    repository.criar(novo);
    return novo;
}

Completar Controllers — POST e PUT reais

CRUD incompleto = -10pts
Todos os seus POSTs retornam uma String hardcoded. Isso não é CRUD. Veja abaixo o padrão que você deve aplicar em cada controller, usando DentistaController como exemplo modelo.
DentistaController.java — versão completa com CRUD real
package br.com.raizdobem.api.controller;

import br.com.raizdobem.api.exception.RecursoNaoEncontradoException;
import br.com.raizdobem.api.entity.Dentista;
import br.com.raizdobem.api.service.DentistaService;
import jakarta.enterprise.context.RequestScoped;
import jakarta.inject.Inject;
import jakarta.transaction.Transactional;
import jakarta.ws.rs.*;
import jakarta.ws.rs.core.MediaType;
import jakarta.ws.rs.core.Response;
import org.eclipse.microprofile.openapi.annotations.Operation;
import org.eclipse.microprofile.openapi.annotations.tags.Tag;
import java.util.List;

@RequestScoped
@Path("/dentista")
@Produces(MediaType.APPLICATION_JSON)
@Consumes(MediaType.APPLICATION_JSON)
@Tag(name = "Dentista", description = "Gerenciamento de dentistas parceiros da ONG.")
public class DentistaController {

    @Inject
    DentistaService service;

    @GET
    @Operation(summary = "Lista todos os dentistas cadastrados.")
    public List<DentistaDTO> listarTodos() {
        return service.listarTodos();
    }

    @GET
    @Path("/{cpf}")
    @Operation(summary = "Busca dentista pelo CPF.")
    public Response buscarPorCpf(@PathParam("cpf") String cpf) {
        return service.buscarPorCpf(cpf)
                .map(d -> Response.ok(d).build())
                .orElseThrow(() -> new RecursoNaoEncontradoException("Dentista", cpf));
    }

    @POST
    @Transactional
    @Operation(summary = "Cadastra novo dentista com validação de CRO.")
    public Response criar(DentistaDTO dto) {
        DentistaDTO criado = service.criar(dto); // valida CRO internamente
        return Response.status(Response.Status.CREATED).entity(criado).build();
    }

    @PUT
    @Path("/{cpf}")
    @Transactional
    @Operation(summary = "Atualiza dados do dentista pelo CPF.")
    public Response atualizar(@PathParam("cpf") String cpf, DentistaDTO dto) {
        DentistaDTO existente = service.buscarPorCpf(cpf)
                .orElseThrow(() -> new RecursoNaoEncontradoException("Dentista", cpf));
        existente.setNomeCompleto(dto.getNomeCompleto());
        existente.setEmail(dto.getEmail());
        existente.setTelefone(dto.getTelefone());
        existente.setDisponivel(dto.getDisponivel());
        return Response.ok(existente).build();
    }

    @DELETE
    @Path("/{cpf}")
    @Transactional
    @Operation(summary = "Remove dentista pelo CPF.")
    public Response excluir(@PathParam("cpf") String cpf) {
        long deletado = service.excluir(cpf);
        if (deletado == 0) throw new RecursoNaoEncontradoException("Dentista", cpf);
        return Response.noContent().build();
    }

    @GET
    @Path("/cidade/{cidade}")
    @Operation(summary = "Lista dentistas por cidade.")
    public List<DentistaDTO> listarPorCidade(@PathParam("cidade") String cidade) {
        return service.listarPorCidades(cidade);
    }

    // ENDPOINT DO MÉTODO DE NEGÓCIO:
    @POST
    @Path("/validar-cro/{cro}")
    @Operation(summary = "Valida se o CRO informado é válido e único no sistema.")
    public Response validarCro(@PathParam("cro") String cro) {
        boolean valido = service.validarCro(cro);
        return Response.ok(Map.of("cro", cro, "valido", valido, "mensagem", "CRO válido e disponível.")).build();
    }
}
